#ler do stdin e numerar o que foi lido(formato "0 ajajaj")
flag = True
res = "" 
n = 0
while flag: 
	stdin = input()
	if stdin == 'quit':
		flag = False
	else:
		aux =str(n).strip() 
		res+=aux+" "+stdin+"\n"
		n+=1
print(res)
